 10 /**
 11  * Envoie une requête XHR
 12  * @param {string} urlSend = URL ou route de l'API
 13  * @return : inutilisable
 14  */
 15  function sendXhrPromise(urlSend){
 16     return new Promise( function(resolve,reject){
 17         console.log("URL : "+ urlSend);
 18         let xhr = new XMLHttpRequest();
 19         xhr.open('GET',urlSend);
 20         xhr.responseType = 'json';
 21         xhr.send();
 22         // Les données arrivent correctement
 23         xhr.addEventListener('load', function(response){
 24             resolve(response.target.response);
 25         })
 26         // Erreur 
 27         xhr.addEventListener('error',function(response){
 28             reject('data transfert error : ' + response);
 29         })
 30     })
 31 }

 33 /**
 34  * On Recupere l'utilisateur que l'on souhaite
 35  * @param {int} uid 
 36  * @returns : inutilisable
 37  */
 38 let getListUsers = (uid) => {
 39     return new Promise ( (resolve, reject) => {
 40         sendXhrPromise('https://api.github.com/users')
 41         .then ( response => {
 42             // Cette variable permet de traiter le cas où on entre un mauvais uid
 43             let userFinded = false;
 44             response.forEach(user => {
 45                 if(user.id === uid){
 46                     userFinded = true;
 47                     resolve(user);
 48                 } 
 49             }); 
 50             if(!userFinded){
 51                 reject("Cet ID n'existe pas");
 52             }
 53         })
 54         .catch ( error => {
 55             console.log(error)
 56         })
 57     })
 58 }

 60 /**
 61  * On inscrit les données de l'utilisateur dans la div prévu à cet effet
 62  * @param {JSON} user 
 63  * @returns : vide
 64  */
 65 let AddUserInfoInHTML = (user) => {
 66     let divResultatUser = document.getElementById('resultatUser');
 67         divResultatUser.innerHTML = "";
 68         htmlAsString = "<div class=\"col-6 text-center mt-4\"><h3> username : " + user.login + "</h3>";
 69         htmlAsString += "<h3> UID : " + user.id + "</h3></div>";
 70         htmlAsString += "<div class=\"col-6 text-center\"><img src="+user.avatar_url + " height=200px width=auto><br></div>";
 71         divResultatUser.innerHTML = htmlAsString;
 72 }

 74 let changeDateFormat = (dateAsString) => {
 75     let date = new Date(dateAsString);
 76     return date.toDateString();
 77 }
 78 
 79 /**
 80  * On remplit le tableau avec les différent repos
 81  * @param {JSON} repos 
 82  */
 83 let AddUserRepoInHTML  = (repos) => {
 84     //Récupération du tableau
 85     let tbody = document.getElementById("tbodyId");
 86     tbody.innerHTML="";
 87     repos.forEach(repo => {
 88         //Ajout d'une ligne dans le tableau concernant les infos du repo
 89         let ajout = "";
 90         ajout+= "<tr>";
 91             ajout+= "<th scope=\"row\">" + repo.name + "</th>";
 92             ajout+= "<td><a href=\""+repo.html_url+"\">"+repo.html_url+"</a></td>";
 93             ajout+= "<td>"+changeDateFormat(repo.updated_at)+"</td>";
 94             ajout+= "<td>"+repo.description+"</td>";
 95             ajout+= "<td>"+repo.visibility+"</td>";
 96         ajout+="</tr>";
 97         tbody.innerHTML+= ajout;
 98     });
 99 }

101 /**
102  * Appel d'une fonction d'affichage des informations de l'utilisateur
103  * @param {*} user 
104  * @returns : inutilisable
105  */
106 let showUserInfo = (user) => {
107     return new Promise( function(resolve,reject){
108         if(user.repos_url){
109             AddUserInfoInHTML(user)
110             resolve(user.repos_url);
111         }
112         else{
113             reject("L'utilisateur n'a pas de lien de repos");
114         }
115     })
116 }

118 /**
119  * Envoie d'une requete et en fonction du resultat on appelera la fonction d'affichage
120  * @param {string} userRepoLink 
121  * @returns : vide
122  */
123 let showUserRepos = (userRepoLink) => {
124     sendXhrPromise(userRepoLink)
125     .then ( repos => {
126         AddUserRepoInHTML(repos)
127     })
128     .catch ( error => {
129         console.log(error)    
130     })
131 }

138 let displayUserRepos = (user_id) => {
